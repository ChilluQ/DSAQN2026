class Solution {
public:
    int hammingWeight(int n) {
         int arr[256];
         int i=0;
         while(n>0){
            arr[i]=n%2;
            i++;
            n=n/2;
         }
        int count=0;
         for(int j =0;j<i;j++){
            if(arr[j]==1){
                count++;
            }
         }
         return count;

    }
};
