class Solution {
  public:
    bool isPalinArray(vector<int> &arr) {
        for(int i =0 ; i<arr.size();i++){
            int n = arr[i];
            int org=n;
            int rev=0;
            while(n>0){
                int last=n%10;
                rev=rev*10+last;
                n/=10;
            }
            if(rev!=org){
                return false;
            }
            return true;
        }
        
    }
};
