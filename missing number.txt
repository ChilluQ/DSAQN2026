class Solution {
public:
    int missingNumber(vector<int>& nums) {
        int n=nums.size();
        int og=0;
        int loc=0;
        for(int i=0;i<n;i++){
            loc+=nums[i];
        }
        
        for(int i=0;i<=n;i++){
            og+=i;
        }

        return og-loc;
    }
};
