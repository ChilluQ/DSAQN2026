class Solution {
  public:
    bool armstrongNumber(int n) {
        int sum=0;
        int og=n;
        
        while(n>0){
            int last=n%10;
            sum+=last*last*last;
            n/=10;
        }
        if(sum!=og){
            return false;
        }
        return true;
        
    }
};
